                    Procedure menu_3;  {comentario}
                    VAR
                       i: integer;
                    begin
                         cursoroff;
                         ClrScr;
                         GotoXY(29,5);
                         ascii('%c', 201);       { ╔ }
                         GotoXY(89,5);
                         ascii('%c', 187);       { ╗ }
                         For i := 30 to 88 do
                         begin
                              GotoXY(i,5);
                              ascii('%c', 205);  { ═ }
                         end;
                         For i := 1 to 6 do
                         begin
                              GotoXY(29,5 + i);
                              ascii('%c', 186);  { ║ }
                              GotoXY(89,5 + i);
                              ascii('%c', 186);  { ║ }
                         end;
                         GotoXY(29,5 + 2);
                         ascii('%c', 204);       { ╠ }
                         GotoXY(89,5 + 2);
                         ascii('%c', 185);       { ╣ }
                         For i := 30 to 88 do
                         begin
                              GotoXY(i,5 + 2);
                              ascii('%c', 205);  { ═ }
                         end;
                         GotoXY(29,5 + 7);
                         ascii('%c', 200);       { ╚ }
                         GotoXY(89,5 + 7);
                         ascii('%c', 188);       { ╝ }
                         For i := 30 to 88 do
                         begin
                              GotoXY(i,5 + 7);
                              ascii('%c', 205);  { ═ }
                         end;
                         Textcolor(14);
                         GotoXY(52,6);
                         Write ('Lista ordenada');
                    end;
